This is the you put your system objetcs. The routes in other words.

Example basico que una ruta con ABM:

<?php

class User extends DomainObject {

    public function __construct() {
        parent::__construct();
        $this->queries['getAll'] = 'user_getAll';
        $this->queries['get'] = 'user_get';
        $this->queries['delete'] = 'user_delete';
        $this->queries['add'] = 'user_add';
        $this->queries['update'] = 'user_update';

    }

}
